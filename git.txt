1.git clone name	--克隆一份代码到本地 name为代码地址

分支操作
2.git branch   		--查看本地分支
3.git branch  -r 	--查看远程分支
4.git branch name	--创建本地分支 name为分支名
5.git checkout name     --切换到name 分支
6.git checkout -b name  --新建分支并切换到新分支
7.git push origin name  --创建远程分支(本地分支push到远程)
8.git branch -d name    --删除分支
9.git merge name        --将name分支合并到当前分支

代码操作
10.git add . 		--将当前目录下修改的所有代码从工作区添加到暂存区 . 代表当前目录
11.git commit -m ‘注释’--将缓存区内容添加到本地仓库
12.git pull  		--提交代码前先更新
13.git push origin name --将本地版本库推送到远程name分支

14.git status		--命令用于显示工作目录和暂存区的状态
15.git log 		--看项目历史的信息要使用

//远程分支拉取到本地分支
git pull origin dev（远程分支）:dev（本地分支）

//git 新版本 pull
git pull origin master --allow-unrelated-histories

//跨分支提交
git config --global push.default upstream

git reset --soft HEAD^  //撤销commit
git push -u origin master -f //强制提交
git config --global credential.helper store //免输入密码
